using System;

class Program
{
    static void Main(string[] args)
    {
        bool equal = Calculator<string>.AreaEqual("Sandip", "Sandip");
        if(equal)
        {
            Console.WriteLine("Equal");
        }
        else
        {
            Console.WriteLine("Not Equal");

        }

    }
}     

public class Calculator<T>
{
    public static bool AreaEqual(T Value1,T Value2)
    {
        return Value1.Equals(Value2);
    }
}
