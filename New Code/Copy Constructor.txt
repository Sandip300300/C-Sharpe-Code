using System;
using System.Collections.Generic;
using System.Collections;
using System.Text;
using System.Threading;

namespace Practice_Project
{
    class Anonymous
    {
        int x;
       public Anonymous(int i)
        {
            x = i;
        }
        public Anonymous(Anonymous obj)
        {
            x = obj.x;
        }
        void display()
        {
            Console.WriteLine(x);
        }
        static void Main(string[] args)
        {
            Anonymous cd1 = new Anonymous(10);
            cd1.display();
            Anonymous cd2 = new Anonymous(cd1);
            cd2.display();


        }
    }
}
